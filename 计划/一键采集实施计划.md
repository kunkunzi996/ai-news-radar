## 1. 目标（做成什么样，用户能验收）

现在用户手动采集要走 4~5 步：① 点「启动抖音采集」→ ② 点「启动小红书采集」→ ③ 干等 + 肉眼核对两张卡都「已完成」→ ④ 点右上角「执行采集」汇总 → 页面重载。

**本次目标**：在「执行采集」旁边加一个「一键采集」按钮。用户点一次，程序自动：

1. 启动抖音采集（需要扫码时用户扫一下）；
2. 抖音跑完后，自动启动小红书采集（需要扫码时用户扫一下）；
3. 两个平台都采集完成后，自动调用现有「执行采集」汇总逻辑并重载页面。

用户全程只在“需要扫码”时动手，其余零操作。

### 已确认的产品决策

- 自动化范围：**全自动**（采集 + 自动收尾汇总，一条龙）。
- 抖音/小红书：**依次跑**（先抖音、跑完再小红书），复用现有同一个浏览器 profile，扫码维护成本最低，后端零改动。

### 本次范围 / 本次不做什么

- ✅ 只加一个前端「一键采集」编排按钮，复用现有接口。
- ❌ 不做“同时跑两个平台”（那需要给两平台分配不同 Chrome 端口和 profile，本次不动）。
- ❌ 不做定时/无人值守自动采集（扫码登录决定了它需要人在场，留待以后）。
- ❌ 不改后端 `scripts/local_server.py`、不改 `scripts/run_mediacrawler_douyin.py`、不改 `scripts/update_news.py`。
- ❌ 不删除、不改动现有卡片上的单平台「启动采集」按钮和「执行采集」按钮（它们作为手动兜底保留）。

**B3. 加编排相关函数。** 放在 `refreshNewsDataFromLocalServer`（`assets/app.js:1695`）**上方**，新增以下三个函数：

```js
const ONE_CLICK_PLATFORM_TIMEOUT_MS = 12 * 60 * 1000; // 单平台最长等 12 分钟（够扫码 + 采集）
const ONE_CLICK_POLL_MS = 3500;                        // 与现有轮询同频

function sleepMs(ms) {
  return new Promise((resolve) => window.setTimeout(resolve, ms));
}

// 启动单个平台采集。复用 /api/maintenance-action，返回 {ok, error}
async function startPlatformCollection(actionId) {
  try {
    const res = await fetch("./api/maintenance-action", {
      method: "POST",
      headers: { "Content-Type": "application/json", Accept: "application/json" },
      body: JSON.stringify({ action_id: actionId, collection_scope: selectedCollectionScope() }),
    });
    const payload = await res.json().catch(() => ({}));
    if (!res.ok || payload.ok === false) {
      return { ok: false, error: payload.error || `HTTP ${res.status}` };
    }
    return { ok: true };
  } catch (err) {
    return { ok: false, error: err?.message || "unknown error" };
  }
}

// 轮询直到该平台“真的这轮跑完”。startedAt 是点“启动”前的时间戳，用来防旧数据误判。
async function waitForCollectorDone(runtimeId, startedAt) {
  const deadline = Date.now() + ONE_CLICK_PLATFORM_TIMEOUT_MS;
  let sawRunning = false;
  // 先给采集进程一点启动时间，避免第一拍还读到上一轮的旧状态
  await sleepMs(ONE_CLICK_POLL_MS);
  while (Date.now() < deadline) {
    const payload = await loadLocalStatusFromServer(false);
    const collector = payload?.collectors?.[runtimeId];
    if (collector) {
      if (collector.running) sawRunning = true;
      const finished = collector.completed && !collector.running;
      const freshMs = collector.updated_at ? Date.parse(collector.updated_at) : 0;
      // 认定“这轮完成”的两个条件（满足其一即可）：
      //  1) 本轮亲眼见它 running 过，且现在已完成；
      //  2) 已完成，且 JSONL 写入时间不早于我们点启动的时间（应对采集极快、没被轮询逮到 running 的情况）。
      if (finished && (sawRunning || (freshMs && freshMs >= startedAt - 1500))) {
        return { done: true };
      }
    }
    await sleepMs(ONE_CLICK_POLL_MS);
  }
  return { done: false, reason: "timeout" };
}

async function runOneClickCollect() {
  if (state.oneClickActive) return;
  state.oneClickActive = true;
  setSourceConfigButton(oneClickCollectBtnEl, "一键采集中…", true);
  setSourceConfigButton(sourceConfigRefreshBtnEl, "执行采集", true); // 编排期间禁用手动收尾，避免并发

  const abort = (msg) => {
    setSourceConfigStatus(`${msg}（可稍后手动点“执行采集”继续）`, "bad");
    restoreSourceConfigButton(oneClickCollectBtnEl, "一键采集");
    restoreSourceConfigButton(sourceConfigRefreshBtnEl, "执行采集");
    state.oneClickActive = false;
    loadLocalStatusFromServer(false);
  };

  try {
    // ① 抖音
    setSourceConfigStatus("① 启动抖音采集…（弹出的采集窗口如提示登录，请扫码）", "warn");
    const dyStart = Date.now();
    const dy = await startPlatformCollection("start_mediacrawler_douyin");
    if (!dy.ok) { abort(`抖音启动失败：${dy.error}`); return; }
    const dyDone = await waitForCollectorDone("mediacrawler_douyin", dyStart);
    if (!dyDone.done) { abort("抖音采集未在规定时间内完成"); return; }

    // ② 小红书
    setSourceConfigStatus("② 抖音已完成，启动小红书采集…（如提示登录，请扫码）", "warn");
    const xhsStart = Date.now();
    const xhs = await startPlatformCollection("start_mediacrawler_xhs");
    if (!xhs.ok) { abort(`小红书启动失败：${xhs.error}`); return; }
    const xhsDone = await waitForCollectorDone("mediacrawler_xhs", xhsStart);
    if (!xhsDone.done) { abort("小红书采集未在规定时间内完成"); return; }

    // ③ 收尾汇总（复用现有逻辑，成功后它会自动重载页面）
    setSourceConfigStatus("③ 两个平台采集完成，正在执行采集汇总…", "warn");
    await refreshNewsDataFromLocalServer();
    // 成功时 refreshNewsDataFromLocalServer 会重载页面，下面这行通常不会执行到
  } catch (err) {
    abort(`一键采集出错：${err?.message || "unknown error"}`);
    return;
  } finally {
    state.oneClickActive = false;
  }
}
```

> 安全要点（务必照做）：
> - **`sawRunning` / `updated_at` 双重判定必须保留**。采集完成状态存在一个坑：上一轮留下的旧 JSONL 也会让 `completed=true`。若不做这个判定，一点按钮就会拿旧数据去汇总。这是本功能最容易出错的地方，标红。
> - **超时兜底必须保留**，否则用户不扫码时按钮会永远转圈。
> - 编排期间禁用「执行采集」手动按钮，`finally` 里复位 `oneClickActive`，避免并发触发两次汇总。

**B4. 绑定按钮。** 在现有 `sourceConfigRefreshBtnEl` 绑定处（`assets/app.js:5175` 附近）旁边加：

```js
if (oneClickCollectBtnEl) {
  oneClickCollectBtnEl.addEventListener("click", runOneClickCollect);
}
```
